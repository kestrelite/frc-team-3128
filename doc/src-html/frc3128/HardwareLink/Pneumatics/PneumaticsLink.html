001package frc3128.HardwareLink.Pneumatics;
002
003import edu.wpi.first.wpilibj.Compressor;
004import edu.wpi.first.wpilibj.Solenoid;
005import frc3128.Util.DebugLog;
006import java.util.Vector;
007
008/**
009 * 
010 * @author Noah Sutton-Smolin
011 */
012public class PneumaticsLink {
013    private   static Compressor c;
014    protected static Vector dualSolenoidList = new Vector();
015
016    /**
017     * Sets up and initialized the PneumaticsManager.
018     * 
019     * @param comp The compressor to be initialized 
020     */
021    public PneumaticsLink(Compressor comp) {
022        PneumaticsLink.c = comp;
023        PneumaticsLink.c.stop();
024    }
025    
026    /**
027     * Sets the compressor. Do not call this unless needed.
028     *
029     * @param comp The compressor to be used
030     */
031    public void setCompressor(Compressor comp) {
032        if(PneumaticsLink.c != null) DebugLog.log(DebugLog.LVL_SEVERE, "PneumaticsLink", "Pneumatic compressor was changed!");
033        PneumaticsLink.c = comp;
034        PneumaticsLink.c.stop();
035    }
036    /**
037     * Adds a piston to the DualSolenoid list
038     * 
039     * @param solA      the first solenoid
040     * @param solB      the second solenoid
041     * @param solAState the first solenoid's initial state
042     * @param solBState the second solenoid's initial state
043     * @return a PistonID for referencing the given piston
044     */
045    public static PistonID addPiston(Solenoid solA, Solenoid solB, boolean solAState, boolean solBState) {
046        dualSolenoidList.addElement(new DualSolenoid(solA, solB, solAState, solBState));
047        return new PistonID(dualSolenoidList.size() - 1);
048    }
049
050    /**
051     * Sets a piston to the locked state (ignores inversion).
052     * 
053     * @param id the PistonID to be set
054     */
055    public static void setPistonStateLocked(PistonID id) {
056        ((DualSolenoid) dualSolenoidList.elementAt(id.getID())).lockPiston();
057    }
058
059    /**
060     * Sets a piston to the unlocked state (ignores inversion).
061     * 
062     * @param id the PistonID to be set
063     */
064    public static void setPistonStateUnlocked(PistonID id) {
065        ((DualSolenoid) dualSolenoidList.elementAt(id.getID())).unlockPiston();
066    }
067
068    /**
069     * Sets a piston to the "on" state ("off" if inverted).
070     * 
071     * @param id the PistonID to be set
072     */
073    public static void setPistonStateOn(PistonID id) {
074        if(!id.getInversion()) 
075            ((DualSolenoid) dualSolenoidList.elementAt(id.getID())).setPistonOn();
076        else
077            ((DualSolenoid) dualSolenoidList.elementAt(id.getID())).setPistonOff();
078    }
079
080    /**
081     * Sets a piston to the "off" state ("on" if inverted).
082     * 
083     * @param id the PistonID to be set
084     */
085    public static void setPistonStateOff(PistonID id) {
086        if(!id.getInversion())
087            ((DualSolenoid) dualSolenoidList.elementAt(id.getID())).setPistonOff();
088        else
089            ((DualSolenoid) dualSolenoidList.elementAt(id.getID())).setPistonOn();
090    }
091
092    /**
093     * Sets a piston to its inverted state (ignores inversion).
094     * 
095     * @param id the PistonID to be inverted
096     */
097    public static void setPistonInvertState(PistonID id) {
098        ((DualSolenoid) dualSolenoidList.elementAt(id.getID())).setPistonInvertState();
099    }
100    
101    /**
102     * Starts the compressor.
103     */
104    public static void setCompressorStateOn() {PneumaticsLink.c.start();}
105
106    /**
107     * Stops the compressor.
108     */
109    public static void setCompressorStateOff() {PneumaticsLink.c.stop();}
110
111    /**
112     * 
113     * @return whether the compressor is on or off
114     */
115    public static boolean getCompressorState() {
116        return (PneumaticsLink.c.enabled() ? true : false);
117    }
118    
119    /**
120     * Locks all pistons.
121     */
122    public static void lockAllPistons() {
123        for(int i = 0; i < dualSolenoidList.size(); i++)
124            ((DualSolenoid) dualSolenoidList.elementAt(i)).lockPiston();
125    }
126}
127
128class DualSolenoid {
129    private Solenoid solA;
130    private Solenoid solB;
131
132    /**
133     * Links two solenoids together, typically a Festo valve.
134     * 
135     * @param solA      the first solenoid valve
136     * @param solB      the second solenoid valve
137     * @param solAState the initial solenoid state for the first valve
138     * @param solBState the initial solenoid state for the second valve
139     */
140    protected DualSolenoid(Solenoid solA, Solenoid solB, boolean solAState, boolean solBState) {
141        this.solA = solA;
142        this.solB = solB;
143        solA.set(solAState);
144        solB.set(solBState);
145    }
146
147    /**
148     * Sets both solenoid states to "true"; locks the piston in place.
149     */
150    protected void lockPiston() {
151        solA.set(true);
152        solB.set(true);
153        DebugLog.log(DebugLog.LVL_STREAM, "DualSolenoid", "DualSolenoid set to lock-state");
154    }
155
156    /**
157     * Sets both solenoid states to "false"; also typically locks the piston in 
158     * place.
159     */
160    protected void unlockPiston() {
161        solA.set(false);
162        solB.set(false);
163        DebugLog.log(DebugLog.LVL_STREAM, "DualSolenoid", "DualSolenoid set to unlock-state");
164    }
165
166    /**
167     * Sets A to true and B to false; generically called "on".
168     */
169    protected void setPistonOn() {
170        solA.set(true);
171        solB.set(false);
172        DebugLog.log(DebugLog.LVL_STREAM, "DualSolenoid", "DualSolenoid set to on-state");
173    }
174
175    /**
176     * Sets B to true and A to false; generically called "off".
177     */
178    protected void setPistonOff() {
179        solB.set(true);
180        solA.set(false);
181        DebugLog.log(DebugLog.LVL_STREAM, "DualSolenoid", "DualSolenoid set to off-state");
182    }
183    
184    /**
185     * Sets B to !B and A to !A, inverting the piston's current state.
186     */
187    protected void setPistonInvertState() {
188        solA.set(!solA.get());
189        solB.set(!solB.get());
190        DebugLog.log(DebugLog.LVL_STREAM, "DualSolenoid", "DualSolenoid set to flip-state " + solA.get() + ", " +solB.get());
191    }
192}
