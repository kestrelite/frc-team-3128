001package frc3128.HardwareLink.Controller;
002
003import edu.wpi.first.wpilibj.Joystick;
004import frc3128.EventManager.Event;
005import frc3128.EventManager.ListenerConst;
006import frc3128.EventManager.ListenerManager;
007import frc3128.Util.DebugLog;
008
009/**
010 * 
011 * @author Noah Sutton-Smolin
012 */
013public class AttackControl extends Event {
014    public  double    x, y, throttle;
015    public  Joystick  aControl;
016    private final int controlID;
017    private boolean[] buttonsPressed = {false, false, false, false, false, false, false, false, false, false, false};
018    
019    /**
020     * Instantiates a new AttackPad controller and starts an Event for updates.
021     * 
022     * @param port the port of the controller
023     */
024    public AttackControl(int port) {
025        aControl = new Joystick(port);
026        controlID = port;
027        this.registerIterableEvent();
028        DebugLog.log(DebugLog.LVL_DEBUG, this, "AttackControl added self to event manager!");
029    }
030    
031    /**
032     * This function will be called automatically by the EventManager.<p><b>Do
033     * not invoke it.</b>
034     */
035    public void execute() {
036        boolean updateJoy = false, updateThrottle = false;
037        
038        if(x != aControl.getAxis(Joystick.AxisType.kX)) updateJoy = true;
039        if(y != aControl.getAxis(Joystick.AxisType.kY)) updateJoy = true;
040        if(throttle != aControl.getAxis(Joystick.AxisType.kThrottle)) updateThrottle = true;
041        
042        x = aControl.getAxis(Joystick.AxisType.kX);
043        y = aControl.getAxis(Joystick.AxisType.kY);
044        throttle = aControl.getAxis(Joystick.AxisType.kThrottle);
045        
046        if(updateJoy) ListenerManager.callListener(ListenerConst.UPDATE_ATK_JOY);
047        if(updateThrottle) ListenerManager.callListener(ListenerConst.UPDATE_ATK_THROTTLE);
048        if(updateJoy || updateThrottle) ListenerManager.callListener(ListenerConst.UPDATE_DRIVE);
049        
050        for(int i = 1; i < 11; i++) {
051            if(buttonsPressed[i] != aControl.getRawButton(i)) {
052                ListenerManager.callListener(ListenerConst.getAtkCtrlListenerKey(this.controlID, i, aControl.getRawButton(i)));
053                DebugLog.log(DebugLog.LVL_STREAM, this, "Button " + (this.controlID + "-" + i) + (aControl.getRawButton(i)==true?" pressed.":" released."));
054            }
055            buttonsPressed[i] = aControl.getRawButton(i);
056        }
057    }
058}
