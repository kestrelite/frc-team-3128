001package frc3128.HardwareLink.Motor;
002
003import frc3128.EventManager.Event;
004
005/**
006 *
007 * @author Noah Sutton-Smolin
008 */
009public abstract class MotorSpeedControl extends Event {
010    private long lastRuntime = -1;
011    private MotorLink controlledMotor = null;
012    
013    public MotorSpeedControl() {}
014    protected void setControlledMotor(MotorLink m) {controlledMotor = m;}
015    
016    public abstract void    setControlTarget(double val);
017    public abstract double  speedTimestep(double dt);
018    public abstract void    clearControlRun();
019    public abstract boolean isComplete();
020    
021    /**
022     * 
023     * @return the last runtime in system clock milliseconds
024     */
025    public final long getLastRuntime() {return lastRuntime;}
026    
027    /**
028     * 
029     * @return how long ago the event was last called (used for dT)
030     */
031    public final long getLastRuntimeDist() {return System.currentTimeMillis() - lastRuntime;}
032
033    /**
034     * 
035     * @return the encoder value of the controlled motor
036     */
037    public double getLinkedEncoderAngle() {return this.controlledMotor.getEncoderAngle();}
038
039    /**
040     * 
041     * @return the linked motor's speed
042     */
043    public double getLinkedMotorSpeed() {return this.controlledMotor.getSpeed();}
044    
045    public final void execute() {
046        lastRuntime = System.currentTimeMillis();
047        this.controlledMotor.setSpeedControlled(this.speedTimestep(this.getLastRuntimeDist()/1000.0));
048        if(this.isComplete()) this.controlledMotor.deleteSpeedControl();
049    }
050}
