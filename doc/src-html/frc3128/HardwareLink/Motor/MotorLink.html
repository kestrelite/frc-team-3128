001package frc3128.HardwareLink.Motor;
002
003import edu.wpi.first.wpilibj.Jaguar;
004import frc3128.HardwareLink.Encoder.AbstractEncoder;
005import frc3128.Util.Constants;
006import frc3128.Util.DebugLog;
007import frc3128.Util.RobotMath;
008
009/**
010 * 
011 * @author Noah Sutton-Smolin
012 */
013//TODO Test MotorLink
014public final class MotorLink {
015        private final Jaguar            motor;
016        private       AbstractEncoder   encoder;
017        private       MotorSpeedControl spdControl;
018        private       boolean           spdControlEnabled = false;
019        private       boolean           motorReversed = false;
020        private       double            powerScalar = 1;
021        
022        /**
023         * 
024         * @param motor the linked motor
025         */
026        public MotorLink(Jaguar motor) {this.motor = motor;}
027        
028        /**
029         * 
030         * @param motor the linked motor
031         * @param enc the linked encoder
032         */
033        public MotorLink(Jaguar motor, AbstractEncoder enc) {this(motor); this.encoder = enc;}
034        
035        /**
036         * 
037         * @param motor the linked motor
038         * @param enc the linked encoder
039         * @param spdControl the linked speed controller
040         */
041        public MotorLink(Jaguar motor, AbstractEncoder enc, MotorSpeedControl spdControl) {this(motor, enc); this.spdControl = spdControl;}
042
043        /**
044         * 
045         * @param motor the linked motor
046         * @param powerScalar the starting power scalar of the motor
047         */
048        public MotorLink(Jaguar motor, double powerScalar) {this(motor); this.setPowerScalar(powerScalar);}
049
050        /**
051         * 
052         * @param motor the linked motor
053         * @param enc the linked encoder
054         * @param powerScalar the starting power scalar of the motor
055         */
056        public MotorLink(Jaguar motor, AbstractEncoder enc, double powerScalar) {this(motor, enc); this.setPowerScalar(powerScalar);}
057        
058        /**
059         * 
060         * @param motor the linked motor
061         * @param enc the linked encoder
062         * @param spdControl the linked speed controller
063         * @param powerScalar the starting power scalar of the motor
064         */
065        public MotorLink(Jaguar motor, AbstractEncoder enc, MotorSpeedControl spdControl, double powerScalar) {this(motor, enc, spdControl); this.setPowerScalar(powerScalar);}
066        
067        /**
068         * 
069         * @return the current speed of the motor
070         */
071        public double getSpeed() {return motor.get();}
072        
073        /**
074         * Sets the speed of the motor. 
075         * 
076         * @param spd the speed to be set
077         */
078        public void setSpeed(double spd) {
079                if(spdControlEnabled) {
080                        DebugLog.log(DebugLog.LVL_WARN, this, "The speed of the motor was set while a controller was active! Deleting speed controller...");
081                        this.deleteSpeedControl();
082                }
083                if(spd < -1 || spd > 1) spd = (spd < 0 ? -1 : 1);
084                
085                this.motor.set(reversedCheck(spd)*this.powerScalar);
086        }
087        protected void setSpeedControlled(double spd) {this.motor.set(spd);}
088        
089        /**
090         * Sets and enables an encoder to be linked with a motor.
091         * 
092         * @param encoder the Encoder to be linked
093         */
094        public void setEncoder(AbstractEncoder encoder) {
095                if(encoder != null) DebugLog.log(DebugLog.LVL_INFO, this, "The encoder was changed in the motor!");
096                this.encoder = encoder;
097        }
098
099        /**
100         * 
101         * @return the current encoder angle
102         */
103        public double getEncoderAngle() {
104                if(encoder == null) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The encoder heading was read, but no encoder was set!"); return -1;}
105                return RobotMath.normalizeAngle(encoder.getAngle());
106        }
107        
108        /**
109         * Clears, then sets a speed controller for the motor. The motor must have
110         * an active encoder.
111         * 
112         * @param spdControl the speed controller to be enabled on the motor
113         */
114        public void loadSpeedControl(MotorSpeedControl spdControl) {
115                if(this.encoder == null) {DebugLog.log(DebugLog.LVL_ERROR, this, "Cannot set the speed controller without an encoder!"); throw new Error();}
116                if(spdControlEnabled) {
117                        DebugLog.log(DebugLog.LVL_SEVERE, this, "The speed controller was set while another was active!");
118                        this.motor.set(0); this.deleteSpeedControl();
119                }
120                this.spdControl = spdControl; this.spdControl.clearControlRun(); 
121                this.spdControlEnabled = true; spdControl.registerIterableEvent();
122        }
123        
124        /**
125         * Deletes the existing speed controller.
126         */
127        public void deleteSpeedControl() {
128                if(spdControl == null) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The speed controller was stopped when it did not exist!"); return;}
129                if(!this.spdControlEnabled) DebugLog.log(DebugLog.LVL_WARN, this, "The speed controller was disabled when it was not active!");
130                this.motor.set(0); this.spdControl.clearControlRun();
131                this.spdControlEnabled = false; spdControl.cancelEvent();
132        }
133        
134        /**
135         * Enables the speed controller on the current motor with the given starting
136         * value.
137         * 
138         * @param targetVal the value to be passed to the speed controller
139         */
140        public void startSpeedControl(double targetVal) {
141                this.spdControl.setControlTarget(targetVal);
142                this.spdControl.clearControlRun();
143                if(spdControl == null) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The speed controller was started when it did not exist!"); return;}
144                this.spdControlEnabled = true; spdControl.registerIterableEvent();
145        }
146        
147        /**
148         * Sets the speed controller's target - will be used however the speed
149         * controller has defined it to be.
150         * 
151         * @param val the value to be sent to the speed controller 
152         */
153        public void setSpeedControlTarget(double val) {
154                if(!this.spdControlEnabled) {
155                        DebugLog.log(DebugLog.LVL_WARN, this, "The speed controller's target was set when it was not active" + (Constants.START_MCONTROL_ON_TARGETSET ? "- enabling..." : "."));
156                        if(Constants.START_MCONTROL_ON_TARGETSET) {this.startSpeedControl(val); return;}
157                }
158                this.spdControl.setControlTarget(val);
159        }
160        
161        /**
162         * Inverts the motor; clockwise becomes counterclockwise and vice versa.
163         */
164        public void reverseMotor() {this.motorReversed = !this.motorReversed;}
165
166        /**
167         * Sets the power scalar for the motor
168         * 
169         * @param pwrScale the power scalar from 0 to 1 to be used for speeds
170         */
171        public void setPowerScalar(double pwrScale) {
172                if(pwrScale < 0 || pwrScale > 1) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The power scalar was set to an illegal value: " + pwrScale + "; scalar not updated!"); return;}
173                this.powerScalar = pwrScale;
174        }
175        
176        /**
177         * 
178         * @return the current power scalar
179         */
180        public double getPowerScalar() {return this.powerScalar;}
181
182        /**
183         * 
184         * @return whether or not the motor is reversed
185         */
186        public boolean getMotorReversed() {return this.motorReversed;}
187        private double reversedCheck(double spd) {return spd * (motorReversed ? -1.0 : 1.0);}
188}
