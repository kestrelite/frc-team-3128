001package frc3128.EventManager;
002
003import frc3128.Util.DebugLog;
004
005/**
006 *
007 * @author Noah Sutton-Smolin
008 */
009final class TimerEvent extends Event {
010    private Event linkedEvent;
011    private long targetTimeMillis = -1;
012
013    public final void setTargetTime(long millis) {
014        DebugLog.log(DebugLog.LVL_STREAM, this, "Event " + linkedEvent.toString() + " set for " + millis + " msec from now.");
015        targetTimeMillis = System.currentTimeMillis() + millis;
016        if(EventManager.containsEvent(this)) {
017            DebugLog.log(DebugLog.LVL_WARN, this, "The timer was set while it was running and before it expired! Deleting old copy...");
018            EventManager.removeEvent(this);
019        }
020        EventManager.addContinuousEvent(this);
021    }
022
023    private void destroyTimer() {
024        targetTimeMillis = -1;
025        EventManager.removeEvent(this);
026    }
027
028    public final void execute() {
029        if (targetTimeMillis == -1) {
030            DebugLog.log(DebugLog.LVL_SEVERE, this, "Timer set without valid time!");
031            this.destroyTimer();
032        }
033
034        if (System.currentTimeMillis() > targetTimeMillis) {
035            DebugLog.log(DebugLog.LVL_STREAM, this, "Running timed event " + this.linkedEvent.toString());
036            linkedEvent.registerSingleEvent();
037            this.destroyTimer();
038        }
039    }
040
041    public final void linkEvent(Event e) {this.linkedEvent = e;}
042}
043
044final class CancelEvent extends Event {
045    private Event linkedEvent;
046    
047    public CancelEvent(Event l) {this.linkedEvent = l;}
048    
049    public void execute() {
050        linkedEvent.cancelEvent();
051        DebugLog.log(DebugLog.LVL_STREAM, this, "Cancelled linked event: " + linkedEvent);
052    }
053}
054
055public abstract class Event {
056    private boolean eventIsCancelled;
057    private TimerEvent timerEvent = null;
058    
059    public Event() {}
060
061    /**
062     * This function is where the event is actually run.
063     */
064    public abstract void execute();
065
066    /**
067     * Cancels the event if it is running. If the event is not running, this
068     * does nothing. <p> Note: This does NOT cancel a timed event.
069     */
070    final public void cancelEvent() {
071        eventIsCancelled = true;
072        DebugLog.log(DebugLog.LVL_STREAM, this, "Event " + this.toString() + " has been cancelled!");
073    }
074
075    /**
076     * Cancels the event after the timer has expired
077     * 
078     * @param msec the time after which the event will cancel
079     */
080    final public void cancelEventAfter(int msec) {
081        DebugLog.log(DebugLog.LVL_STREAM, this, "Canceling self by trigger after " + msec + " msec.");
082        new CancelEvent(this).registerTimedEvent(msec);
083    }
084    
085    /**
086     * Cancels the event's timer if it is running. If the timer is not running,
087     * this does nothing. <p> Note: This does NOT cancel the event's execution.
088     */
089    final public void cancelTimedEvent() {
090        timerEvent.cancelEvent();
091    }
092
093    final protected boolean shouldRun() {
094        return !eventIsCancelled;
095    }
096
097    /**
098     * Adds the event to the EventManager's queue as a single-run event. This
099     * event will be run once, and then removed from the queue.
100     */
101    final public void registerSingleEvent() {
102        this.eventIsCancelled = false;
103        EventManager.addSingleEvent(this);
104    }
105
106    /**
107     * Adds the event to the EventManager's queue as an iterable event. This
108     * event will not be deleted from the queue once it has been run; it will be
109     * run every iteration. It must be canceled explicitly.
110     */
111    final public void registerIterableEvent() {
112        this.eventIsCancelled = false;
113        EventManager.addContinuousEvent(this);
114    }
115
116    /**
117     * Starts the event's linked TimerEvent. The TimerEvent is an iterable
118     * event. Each iteration the TimerEvent will check to see if the time has
119     * expired; if it has, it will run the current event as a SingleEvent. <p>
120     * Note: If the timer has not been created, then it will be created when
121     * this method is first called. 
122     *
123     * @param msec the time in milliseconds after which the event will execute.
124     */
125    final public void registerTimedEvent(int msec) {
126        if (timerEvent == null) {
127            this.timerEvent = new TimerEvent();
128            timerEvent.linkEvent(this);
129        }
130        timerEvent.setTargetTime(msec);
131    }
132}
