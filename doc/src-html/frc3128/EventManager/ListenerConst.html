001package frc3128.EventManager;
002
003import frc3128.HardwareLink.Controller.XControlMap;
004
005/**
006 *
007 * @author Noah Sutton-Smolin
008 */
009public class ListenerConst {
010    public static final String UPDATE_JOY1 = "updateJoy1";
011    public static final String UPDATE_JOY2 = "updateJoy2";
012    public static final String UPDATE_TRIGGERS = "updateTriggers";
013    public static final String UPDATE_DRIVE = "updateDrive";
014    
015    public static final String UPDATE_ATK_JOY = "updateAtkJoy";
016    public static final String UPDATE_ATK_THROTTLE = "updateAtkThrottle";
017    
018    /**
019     * Builds a string for the AttackPad controller listeners.
020     * 
021     * @param controllerPort the port number of the controller
022     * @param buttonNum      the button number pressed
023     * @param pressed        true if down, false if up
024     * @return a listener for ListenerManager which AttackControl will trigger
025     */
026    public static String getAtkCtrlListenerKey(int controllerPort, int buttonNum, boolean pressed) {
027        return "button" + controllerPort + "-" + buttonNum + (pressed?"Down":"Up");
028    }
029    
030    /**
031     * Builds a string for the XBox controller listeners.
032     * 
033     * @param controllerPort the port number of the controller
034     * @param buttonNum the button number pressed
035     * @param pressed true if down, false if up
036     * @return a listener for ListenerManager which XControl will trigger
037     */
038    public static String getXCtrlListenerKey(int controllerPort, int buttonNum, boolean pressed) {
039        return "button" + controllerPort + "-" + XControlMap.getBtnString(buttonNum) + (pressed?"Down":"Up");
040    }
041    
042    private ListenerConst() {}
043}
