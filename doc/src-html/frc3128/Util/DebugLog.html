001package frc3128.Util;
002
003/**
004 *
005 * @author Noah Sutton-Smolin
006 */
007public class DebugLog {
008    public static final int LVL_STREAM = 5;
009    public static final int LVL_DEBUG = 4;
010    public static final int LVL_INFO = 3;
011    public static final int LVL_WARN = 2;
012    public static final int LVL_SEVERE = 1;
013    public static final int LVL_ERROR = 0;
014    private static int logDetail = Constants.DEBUGLOG_DEFAULT_LOGLEVEL;
015    private static int maxTagLength = 0;
016    private static final int initTagLength = 32;
017    private static int skippedItems = 0;
018    private static int totalItems = 0;
019    private static int[] totalItemsType = new int[5];
020    private static double wastedTimeAll = 0;
021    private static double wastedTimeText = 0;
022    private static double startTime = -1;
023
024    /**
025     * This sets the log detail level. This is used to filter based on severity.
026     * Severity is the following: <p>0: Error; 1: Severe; 2: Warning; 3: Info;
027     * 4: Debug; 5: Stream
028     *
029     * @param level the level to be logged.
030     */
031    public static void setLogLevel(int level) {
032        DebugLog.log(DebugLog.LVL_INFO, "DebugLog", "Log detail set to " + level);
033        DebugLog.logDetail = level;
034    }
035
036    /**
037     * Logs a message to the NetBeans console. The log keeps all statements in
038     * alignment with the longest message. It will hide any information whose
039     * log level is higher than the provided maximum detail. It uses an object
040     * (typically a string) to create a tag for the message.
041     *
042     * @param level the severity level of the message to be logged.
043     * @param obj the object (typically String) used to identify the message
044     * source.
045     * @param text the message to be logged.
046     */
047    public static void log(int level, Object obj, String text) {
048                if(!Constants.DEBUGLOG_ENABLED) return;
049                
050        long logStartTime = System.currentTimeMillis(); //Diagnostic
051
052        String strLv = "[UNKN]  ";
053        if (level <= 0) strLv = "[ERROR" + Math.abs(level) + "] ";
054        if (level == 1) strLv = "[SEVERE]";
055        if (level == 2) strLv = "[**WARN]";
056        if (level == 3) strLv = "[INFO]  ";
057        if (level == 4) strLv = "[DEBUG] ";
058        if (level == 5) strLv = "[STREAM]";
059
060        if (obj.toString().length() > DebugLog.maxTagLength && level <= DebugLog.logDetail)
061            DebugLog.maxTagLength = obj.toString().length();
062
063        if (level <= DebugLog.logDetail) {
064            double textStartTime = System.currentTimeMillis(); //Diagnostic
065
066            System.out.print("[" + System.currentTimeMillis() + "] "
067                    + strLv + " ["
068                    + (obj.toString().substring(0, 4).equals("edu.") ? obj.toString().substring(initTagLength) : obj.toString()) + "] ");
069            for (int i = 0; i < maxTagLength - obj.toString().length(); i++)
070                System.out.print(" ");
071            System.out.println(text);
072
073            DebugLog.wastedTimeText += (System.currentTimeMillis() - textStartTime); //Diagnostic
074        } else DebugLog.skippedItems++; //From this line forward, begin diagnostics
075        DebugLog.totalItems++;
076        DebugLog.totalItemsType[level - 1]++;
077
078        if (DebugLog.totalItems % Constants.DEBUGLOG_INFO_DISPLAYFREQ == 0) {
079            int currLogLevel = DebugLog.logDetail;
080            DebugLog.logDetail = 3;
081            DebugLog.log(DebugLog.LVL_INFO, "DebugLog", "DebugLog has skipped " + skippedItems + ", has displayed " + totalItems + " over " + ((System.currentTimeMillis() - DebugLog.startTime) / 1000.0) + " msec.");
082            DebugLog.log(DebugLog.LVL_INFO, "DebugLog", "Breaking down by type (0...5): " + "\n\t\tError: " + DebugLog.totalItemsType[0]
083                    + "\n\t\tSevere: " + DebugLog.totalItemsType[1]
084                    + "\n\t\tWarning: " + DebugLog.totalItemsType[2]
085                    + "\n\t\tInfo: " + DebugLog.totalItemsType[3]
086                    + "\n\t\tDebug: " + DebugLog.totalItemsType[4]);
087            DebugLog.log(DebugLog.LVL_INFO, "DebugLog", "DebugLog averages " + (totalItems / ((System.currentTimeMillis() - DebugLog.startTime) / 1000.0)) + " messages per second.");
088            DebugLog.log(DebugLog.LVL_INFO, "DebugLog", "DebugLog has wasted an approximate total of " + (DebugLog.wastedTimeAll / 1000.0) + " seconds, " + (DebugLog.wastedTimeText / 1000.0) + " of which was spent rendering text.");
089            DebugLog.log(DebugLog.LVL_INFO, "DebugLog", "If this is considerably too high, then please consider removing stream messages.");
090            DebugLog.logDetail = currLogLevel;
091        }
092        DebugLog.wastedTimeAll += (System.currentTimeMillis() - logStartTime);
093        //End diagnostic code
094    }
095
096    private DebugLog() {}
097}
