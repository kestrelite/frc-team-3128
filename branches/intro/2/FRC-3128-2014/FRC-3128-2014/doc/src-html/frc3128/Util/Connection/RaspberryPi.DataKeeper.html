149        public synchronized String getRawData() throws IOException {
150        byte[] input;
151
152        if (connected) {
153            outStream.write('G'); //request Data
154            DebugLog.log(DebugLog.LVL_DEBUG, this, "Requested Data");
155
156                        if (inStream.available() <= BUFFERSIZE) {
157                                input = new byte[inStream.available()]; //storage space sized to fit!
158                                receivedData = new byte[inStream.available()];
159                                inStream.read(input);
160                                for (int i = 0; (i < input.length) && (input != null); i++)
161                                        receivedData[i] = input[i]; //transfer input to full size storage
162                        } else {
163                                DebugLog.log(DebugLog.LVL_WARN, this, "Data buffer overflow!");
164                                inStream.skip(inStream.available()); //reset if more is stored than buffer
165                                return null;
166                        }
167
168            rawData = ""; //String to transfer received data to
169                        DebugLog.log(DebugLog.LVL_DEBUG, this, "Raw data size: " + receivedData.length);
170            for (int i = 0; i < receivedData.length; i++)
171                rawData += (char) receivedData[i]; //Cast bytes to chars and concatenate them to the String\
172                        DebugLog.log(DebugLog.LVL_DEBUG, this, "Raw data: " + rawData);
173            return rawData;
174        } else {connect(); return null;}
175    }

177        public synchronized String[] tokenizeData(String input) {
178        StringTokenizer tokenizer = new StringTokenizer(input, String.valueOf(DELIMITER));
179        String output[] = new String[tokenizer.countTokens()];
180                
181                for (int i = 0; i < output.length; i++)
182            output[i] = tokenizer.nextToken();
183        return output;
