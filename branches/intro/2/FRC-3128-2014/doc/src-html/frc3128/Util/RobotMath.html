001package frc3128.Util;
002
003import frc3128.HardwareLink.Motor.MotorDir;
004
005/**
006 *
007 * @author Noah Sutton-Smolin
008 */
009public class RobotMath {
010        /**
011         * Limits the angle to between 0 and 359 degrees for all math. All angles
012         * should be normalized before use.
013         * 
014         * @param angle the angle to be normalized
015         * @return the normalized angle on [0,359]
016         */
017        public static double normalizeAngle(double angle) {
018                return ((angle % 360) + 360) % 360;
019        }
020        
021        /**
022         * 
023         * @param pow motor power
024         * @return whether or not the power is valid
025         */
026        public static boolean isValidPower(double pow) {return (pow >= -1 && pow <= 0);}
027        
028        /**
029         * Makes the provided power into a valid power level.
030         * 
031         * @param pow power level to convert
032         * @return a properly-limited power level
033         */
034        public static double makeValidPower(double pow) {return (pow < -1 ? -1 : (pow > 1 ? 1 : pow));}
035        
036        /**
037         * Determines the appropriate direction for a motor to turn.
038         * 
039         * @param currentAngle the current angle of the motor
040         * @param targetAngle the target angle of the motor
041         * @return an integer value on {-1,0,1} in MotorDir
042         */
043        public static int getMotorDirToTarget(double currentAngle, double targetAngle) {
044                currentAngle = RobotMath.normalizeAngle(currentAngle);
045                targetAngle = RobotMath.normalizeAngle(targetAngle);
046                int retDir = 1 * (Math.abs(currentAngle - targetAngle) > 180 ? 1 : -1) * (currentAngle - targetAngle < 0 ? -1 : 1);
047                
048                //MotorDir is syntactic sugar; could just return retDir
049                if(currentAngle - targetAngle == 0 || currentAngle - targetAngle == 180) return MotorDir.EITHER;
050                return (retDir == 1 ? MotorDir.CW : MotorDir.CCW); 
051        }
052        
053        private RobotMath() {}
054}
