001package frc3128.EventManager;
002
003import frc3128.Util.DebugLog;
004
005/**
006 *
007 * @author Noah Sutton-Smolin
008 */
009final class TimerEvent extends Event {
010    private Event linkedEvent;
011    private long targetTimeMillis = -1;
012
013    public final void setTargetTime(long millis) {
014        DebugLog.log(DebugLog.LVL_STREAM, this, "Event " + linkedEvent.toString() + " set for " + millis + " msec from now.");
015        targetTimeMillis = System.currentTimeMillis() + millis;
016        EventManager.addContinuousEvent(this);
017    }
018
019    private void destroyTimer() {
020        targetTimeMillis = -1;
021        EventManager.removeEvent(this);
022    }
023
024    public final void execute() {
025        if (targetTimeMillis == -1) {
026            DebugLog.log(DebugLog.LVL_SEVERE, this, "Timer set without valid time!");
027            this.destroyTimer();
028        }
029
030        if (System.currentTimeMillis() > targetTimeMillis) {
031            DebugLog.log(DebugLog.LVL_STREAM, this, "Running timed event " + this.linkedEvent.toString());
032            linkedEvent.registerSingleEvent();
033            this.destroyTimer();
034        }
035    }
036
037    public final void linkEvent(Event e) {
038        this.linkedEvent = e;
039    }
040}
041
042final class CancelEvent extends Event {
043    private Event linkedEvent;
044    
045    public CancelEvent(Event l) {this.linkedEvent = l;}
046    
047    public void execute() {
048        linkedEvent.cancelEvent();
049        DebugLog.log(DebugLog.LVL_DEBUG, this, "Cancelled linked event: " + linkedEvent);
050    }
051}
052
053public abstract class Event {
054    private boolean eventIsCancelled;
055    private TimerEvent timerEvent = null;
056    private long cancelAtTime = -1;
057    
058    public Event() {
059    }
060
061    public Event(boolean isTimerEvent) {
062        if (isTimerEvent) {
063            this.timerEvent = new TimerEvent();
064            timerEvent.linkEvent(this);
065        }
066    }
067
068    /**
069     * This function is where the event is actually run.
070     */
071    public abstract void execute();
072
073    /**
074     * Cancels the event if it is running. If the event is not running, this
075     * does nothing. <p> Note: This does NOT cancel a timed event.
076     */
077    final public void cancelEvent() {
078        eventIsCancelled = true;
079        EventManager.removeEvent(this);
080        DebugLog.log(DebugLog.LVL_DEBUG, this, "Event " + this.toString() + " has been cancelled!");
081    }
082
083    final public void cancelEventAfter(int msec) {
084        DebugLog.log(DebugLog.LVL_DEBUG, this, "Canceling self by trigger after " + msec + " msec.");
085        new CancelEvent(this).registerTimedEvent(msec);
086    }
087    
088    /**
089     * Cancels the event's timer if it is running. If the timer is not running,
090     * this does nothing. <p> Note: This does NOT cancel the event's execution.
091     */
092    final public void cancelTimedEvent() {
093        timerEvent.cancelEvent();
094    }
095
096    final protected boolean shouldRun() {
097        return !eventIsCancelled;
098    }
099
100    /**
101     * Adds the event to the EventManager's queue as a single-run event. This
102     * event will be run once, and then removed from the queue.
103     */
104    final public void registerSingleEvent() {
105        EventManager.addSingleEvent(this);
106    }
107
108    /**
109     * Adds the event to the EventManager's queue as an iterable event. This
110     * event will not be deleted from the queue once it has been run; it will be
111     * run every iteration. It must be canceled explicitly.
112     */
113    final public void registerIterableEvent() {
114        EventManager.addContinuousEvent(this);
115    }
116
117    /**
118     * Starts the event's linked TimerEvent. The TimerEvent is an iterable
119     * event. Each iteration the TimerEvent will check to see if the time has
120     * expired; if it has, it will run the current event as a SingleEvent. <p>
121     * Note: If the timer has not been created, then it will be created when
122     * this method is first called. It is recommended that you call
123     * Event.prepareTimer() before executing this method.
124     *
125     * @param msec the time in milliseconds after which the event will execute.
126     */
127    final public void registerTimedEvent(int msec) {
128        if (timerEvent == null) {
129            DebugLog.log(DebugLog.LVL_WARN, this, "Timer event called before instantiation! Timer startup delay possible.");
130            prepareTimer();
131        }
132        timerEvent.setTargetTime(msec);
133    }
134
135    /**
136     * Creates and links an instance of the TimerEvent. The TimerEvent must be
137     * created if Event.registerTimedEvent() is to be called. It is recommended
138     * that you call this function before invoking registerTimedEvent(). <p> You
139     * may also pass "true" to the constructor during instantiation to create
140     * the TimerEvent.
141     */
142    final public void prepareTimer() {
143        if (this.timerEvent != null) return;
144        this.timerEvent = new TimerEvent();
145        timerEvent.linkEvent(this);
146    }
147}
