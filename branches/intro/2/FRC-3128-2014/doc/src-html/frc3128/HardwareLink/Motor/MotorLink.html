001package frc3128.HardwareLink.Motor;
002
003import edu.wpi.first.wpilibj.Jaguar;
004import frc3128.HardwareLink.Encoder.AbstractEncoder;
005import frc3128.Util.Constants;
006import frc3128.Util.DebugLog;
007import frc3128.Util.RobotMath;
008
009/**
010 * 
011 * @author Noah Sutton-Smolin
012 */
013//TODO Test MotorLink
014public class MotorLink {
015        private final Jaguar            motor;
016        private       AbstractEncoder   encoder;
017        private       MotorSpeedControl spdControl;
018        private       boolean           spdControlEnabled = false;
019        private       boolean           motorReversed = false;
020        private       double            powerScalar = 1;
021        
022        /**
023         * 
024         * @param motor the linked motor
025         */
026        public MotorLink(Jaguar motor) {this.motor = motor;}
027        
028        /**
029         * 
030         * @param motor the linked motor
031         * @param enc the linked encoder
032         */
033        public MotorLink(Jaguar motor, AbstractEncoder enc) {this(motor); this.encoder = enc;}
034        
035        /**
036         * 
037         * @param motor the linked motor
038         * @param enc the linked encoder
039         * @param spdControl the linked speed controller
040         */
041        public MotorLink(Jaguar motor, AbstractEncoder enc, MotorSpeedControl spdControl) {this(motor, enc); this.spdControl = spdControl;}
042        
043        /**
044         * 
045         * @return the current speed of the motor
046         */
047        public double getSpeed() {return motor.get();}
048        
049        /**
050         * Sets the speed of the motor. 
051         * 
052         * @param spd the speed to be set
053         */
054        public void setSpeed(double spd) {
055                if(spdControlEnabled) {
056                        DebugLog.log(DebugLog.LVL_WARN, this, "The speed of the motor was set while a controller was active! Deleting speed controller...");
057                        this.deleteSpeedControl();
058                }
059                if(spd < -1 || spd > 1) spd = (spd < 0 ? -1 : 1);
060                
061                this.motor.set(reversedCheck(spd)*this.powerScalar);
062        }
063        protected void setSpeedControlled(double spd) {this.motor.set(spd);}
064        
065        /**
066         * Sets and enables an encoder to be linked with a motor.
067         * 
068         * @param encoder the Encoder to be linked
069         */
070        public void setEncoder(AbstractEncoder encoder) {
071                if(encoder != null) DebugLog.log(DebugLog.LVL_INFO, this, "The encoder was changed in the motor!");
072                this.encoder = encoder;
073        }
074
075        /**
076         * 
077         * @return the current encoder angle
078         */
079        public double getEncoderAngle() {
080                if(encoder == null) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The encoder heading was read, but no encoder was set!"); return -1;}
081                return RobotMath.normalizeAngle(encoder.getAngle());
082        }
083        
084        /**
085         * Clears, then sets a speed controller for the motor. The motor must have
086         * an active encoder.
087         * 
088         * @param spdControl the speed controller to be enabled on the motor
089         */
090        public void loadSpeedControl(MotorSpeedControl spdControl) {
091                if(this.encoder == null) {DebugLog.log(DebugLog.LVL_ERROR, this, "Cannot set the speed controller without an encoder!"); throw new Error();}
092                if(spdControlEnabled) {
093                        DebugLog.log(DebugLog.LVL_SEVERE, this, "The speed controller was set while another was active!");
094                        this.motor.set(0); this.deleteSpeedControl();
095                }
096                this.spdControl = spdControl; this.spdControl.clearControlRun(); 
097                this.spdControlEnabled = true; spdControl.registerIterableEvent();
098        }
099        
100        /**
101         * Deletes the existing speed controller.
102         */
103        public void deleteSpeedControl() {
104                if(spdControl == null) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The speed controller was stopped when it did not exist!"); return;}
105                if(!this.spdControlEnabled) DebugLog.log(DebugLog.LVL_WARN, this, "The speed controller was disabled when it was not active!");
106                this.motor.set(0); this.spdControl.clearControlRun();
107                this.spdControlEnabled = false; spdControl.cancelEvent();
108        }
109        
110        /**
111         * Enables the speed controller on the current motor with the given starting
112         * value.
113         * 
114         * @param targetVal the value to be passed to the speed controller
115         */
116        public void startSpeedControl(double targetVal) {
117                this.spdControl.setControlTarget(targetVal);
118                this.spdControl.clearControlRun();
119                if(spdControl == null) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The speed controller was started when it did not exist!"); return;}
120                this.spdControlEnabled = true; spdControl.registerIterableEvent();
121        }
122        
123        /**
124         * Sets the speed controller's target - will be used however the speed
125         * controller has defined it to be.
126         * 
127         * @param val the value to be sent to the speed controller 
128         */
129        public void setSpeedControlTarget(double val) {
130                if(!this.spdControlEnabled) {
131                        DebugLog.log(DebugLog.LVL_WARN, this, "The speed controller's target was set when it was not active" + (Constants.START_MCONTROL_ON_TARGETSET ? "- enabling..." : "."));
132                        if(Constants.START_MCONTROL_ON_TARGETSET) {this.startSpeedControl(val); return;}
133                }
134                this.spdControl.setControlTarget(val);
135        }
136        
137        /**
138         * Inverts the motor; clockwise becomes counterclockwise and vice versa.
139         */
140        public void reverseMotor() {this.motorReversed = !this.motorReversed;}
141
142        /**
143         * Sets the power scalar for the motor
144         * 
145         * @param pwrScale the power scalar from 0 to 1 to be used for speeds
146         */
147        public void setPowerScalar(double pwrScale) {
148                if(pwrScale < 0 || pwrScale > 1) {DebugLog.log(DebugLog.LVL_SEVERE, this, "The power scalar was set to an illegal value: " + pwrScale + "; scalar not updated!"); return;}
149                this.powerScalar = pwrScale;
150        }
151        
152        /**
153         * 
154         * @return the current power scalar
155         */
156        public double getPowerScalar() {return this.powerScalar;}
157
158        
159        /**
160         * 
161         * @return whether or not the motor is reversed
162         */
163        public boolean getMotorReversed() {return this.motorReversed;}
164        private double reversedCheck(double spd) {return spd * (motorReversed ? -1.0 : 1.0);}
165}
