010//TODO Test LinearAngleTarget
011public class LinearAngleTarget extends MotorSpeedControl {
012        private double minSpeed;
013        private double targetAngle, tolerance;
014        
015        public LinearAngleTarget(double minSpeed, double tolerance) {
016                if(!RobotMath.isValidPower(minSpeed)) throw new IllegalArgumentException("The minimum power is incorrect!"); 
017                this.minSpeed = Math.abs(minSpeed);
018        }
019        
020        public void setControlTarget(double val) {this.targetAngle = val;}
021
022        public double speedTimestep(double dt) {
023                double currentAngle = RobotMath.normalizeAngle(this.getLinkedEncoderAngle());
024                targetAngle = RobotMath.normalizeAngle(this.getLinkedEncoderAngle());
025                double angleDiff = RobotMath.normalizeAngle(Math.abs(currentAngle - targetAngle));
026                return (angleDiff / 180.0) * ((1.0 - minSpeed)/1.0) + minSpeed;
027        }
028
029        public void clearControlRun() {this.targetAngle = 0;}
030        
031        public boolean isComplete() {return Math.abs(this.getLinkedEncoderAngle() - targetAngle) < tolerance;}
032}
