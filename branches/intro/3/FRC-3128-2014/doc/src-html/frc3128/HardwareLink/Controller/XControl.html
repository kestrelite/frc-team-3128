001package frc3128.HardwareLink.Controller;
002
003import edu.wpi.first.wpilibj.Joystick;
004import frc3128.EventManager.Event;
005import frc3128.EventManager.ListenerManager;
006import frc3128.Util.DebugLog;
007
008/**
009 * 
010 * @author Noah Sutton-Smolin
011 */
012public class XControl extends Event {
013    private Joystick xControl;
014    public double x1, y1, x2, y2, triggers;
015    private boolean[] buttonsPressed = {false, false, false, false, false, false, false, false, false, false, false};
016    
017        /**
018         * Instantiates a new XBox controller and starts an Event for updates.
019         * 
020         * @param port the port of the controller
021         */
022    public XControl(int port) {
023        xControl = new Joystick(port);
024        this.registerIterableEvent();
025        DebugLog.log(DebugLog.LVL_DEBUG, this, "XBox Controller added self to event manager!");
026    }
027
028        /**
029         * This function will be called automatically by the EventManager.<p><b>Do
030         * not invoke it.</b>
031         */
032    public void execute() {
033        boolean updateJoy1 = false;
034        boolean updateJoy2 = false;
035        boolean updateTriggers = false;
036
037        if(x1 != xControl.getRawAxis(XControlMap.X1_AXIS))       updateJoy1 = true;
038        if(x2 != xControl.getRawAxis(XControlMap.X2_AXIS))       updateJoy2 = true;
039        if(y1 != xControl.getRawAxis(XControlMap.Y1_AXIS))       updateJoy1 = true;
040        if(y2 != xControl.getRawAxis(XControlMap.Y2_AXIS))       updateJoy2 = true;
041        if(triggers != xControl.getRawAxis(XControlMap.TRIGGER_AXIS)) updateTriggers = true;
042
043        x1 = xControl.getRawAxis(XControlMap.X1_AXIS);
044        x2 = xControl.getRawAxis(XControlMap.X2_AXIS);
045        y1 = xControl.getRawAxis(XControlMap.Y1_AXIS);
046        y2 = xControl.getRawAxis(XControlMap.Y2_AXIS);
047
048        if(updateJoy1) ListenerManager.callListener("updateJoy1");
049        if(updateJoy2) ListenerManager.callListener("updateJoy2");
050        if(updateTriggers) ListenerManager.callListener("updateTriggers");
051        if(updateJoy1 || updateJoy2 || updateTriggers) ListenerManager.callListener("updateDrive");
052
053        for(int i = 1; i < 11; i++) {
054            if(buttonsPressed[i] != xControl.getRawButton(i)) {
055                ListenerManager.callListener("button" + XControlMap.getBtnString(i) + (xControl.getRawButton(i) ? "Down" : "Up"));
056                DebugLog.log(DebugLog.LVL_STREAM, this, "Button " + XControlMap.getBtnString(i) + (xControl.getRawButton(i)==true?" pressed.":" released."));
057            }
058            buttonsPressed[i] = xControl.getRawButton(i);
059        }
060    }
061}
