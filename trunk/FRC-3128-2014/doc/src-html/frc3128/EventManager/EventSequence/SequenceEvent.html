001package frc3128.EventManager.EventSequence;
002
003import frc3128.EventManager.Event;
004
005/**
006 * 
007 * @author Noah Sutton-Smolin
008 */
009public abstract class SequenceEvent extends Event {
010    private long    startTime      = -1;
011    private boolean eventIsRunning = false;
012    
013    public SequenceEvent() {super();}
014    
015        /**
016         * Starts the sequence event running. 
017         */
018    final protected void startSequenceEvent() {
019        if(startTime == -1) this.startTime = System.currentTimeMillis();
020        this.eventIsRunning = true;
021    }
022        
023        /**
024         * Returns whether or not the current SequenceEvent is running.
025         * 
026         * @return whether the current SequenceEvent is running
027         */
028    final protected boolean isRunning() {return eventIsRunning;}
029    
030        /**
031         * Finds how long the SequenceEvent ran.
032         * 
033         * @return the duration, in msec, for how long the SequenceEvent ran
034         */
035        final protected long getRunTimeMillis() {return (startTime == -1 ? -1 : System.currentTimeMillis() - this.startTime);}
036
037        /**
038         * 
039         * @return whether the current SequenceEvent's exit condition has been met
040         */
041        public abstract boolean exitConditionMet();
042}
