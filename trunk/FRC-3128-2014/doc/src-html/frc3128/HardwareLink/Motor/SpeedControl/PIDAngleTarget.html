010public class PIDAngleTarget extends MotorSpeedControl {

012        private final double secs;
013        private       double targetAngle, threshold;

015        public PIDAngleTarget(double targetAngle, double threshold, int msec, int kP, int kI, int kD) {
016                this.kP = kP; this.kI = kI; this.kD = kD;
017                this.secs = ((double)msec)/1000.0;
018                this.targetAngle = targetAngle; this.threshold = threshold;
019        }
020        
021        public void setControlTarget(double d) {}
022
023        public double speedTimestep(double dt) {return 0;}
024
025        public void clearControlRun() {}
026        
027        public boolean isComplete() {return Math.abs(this.getLinkedEncoderAngle() - this.targetAngle)<threshold;}
